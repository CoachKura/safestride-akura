const { data: { user } } = await supabaseClient.auth.getUser(); if (!user) { console.log('❌ No user session found'); return null; } // Query strava_connections table const { data, error } = await supabaseClient .from('strava_connections') .select('*') .eq('athlete_id', user.id) .single(); if (error) { console.log('ℹ️ No Strava connection found:', error.message); return null; } console.log('✅ Found existing Strava connection:', data); // Check if token expired const expiresAt = new Date(data.expires_at); const now = new Date(); if (expiresAt < now) { console.log('⚠️ Token expired, refreshing...'); const refreshed = await refreshStravaToken(data.athlete_id, data.refresh_token); if (!refreshed) return null; return refreshed; } // Update global state isStravaConnected = true; stravaProfile = { strava_athlete_id: data.strava_athlete_id, strava_access_token: data.access_token, athlete_data: data.athlete_data }; return data; } catch (error) { console.error('❌ Error checking connection:', error); return null; } } async function refreshStravaToken(athleteId, refreshToken) { try { console.log('🔄 Refreshing Strava token...'); const response = await fetch('https://bdisppaxbvygsspcuymb.supabase.co/functions/v1/strava-refresh-token', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${SUPABASE_ANON_KEY}` }, body: JSON.stringify({ athleteId, refreshToken }) }); if (!response.ok) throw new Error('Refresh failed'); const result = await response.json(); console.log('✅ Token refreshed'); return result.data; } catch (error) { console.error('❌ Token refresh error:', error); return null; } } function updateStravaConnectionUI(connectionData) { const connectBtn = document.getElementById('connectStravaBtn'); const statusDiv = document.getElementById('strava-status'); if (!connectBtn) return; const daysAgo = (Date.now() - activityDate.getTime()) / (1000 * 60 * 60 * 24); return daysAgo <= 30; }); athleteData.recentActivityCount = last30Days.length; athleteData.totalDistance = last30Days.reduce((sum, a) => sum + (a.distance || 0), 0) / 1000; if (error) { console.error('❌ Error loading scores:', error); return; } if (data.length > 0) { console.log('✅ Loaded AISRI score:', data[0]); athleteData.aisriScore = data[0]; } } catch (error) { console.error('❌ Error loading scores:', error); } } // Check for Strava OAuth callback and connection status window.addEventListener("load", async function () { console.log('🚀 AISRI Training Plan Builder loaded'); // Check existing connection FIRST const existingConnection = await checkExistingStravaConnection();
